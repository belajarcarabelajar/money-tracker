            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    // Fallback ke Login Anonim agar database tetap bisa ditulis (write access)
                    await signInAnonymously(auth);
                }
            } catch (error) {
                console.error("Auth Error:", error);
                if(statusText) statusText.textContent = "Gagal Terhubung";
                if(statusDot) { statusDot.classList.remove('connecting'); statusDot.style.backgroundColor = 'var(--color-danger)'; }
                if (typeof showToast === 'function') showToast("Gagal login: " + error.message, "error");
            }

                    let finalTranscript = '';
                    for (let i = event.resultIndex; i < event.results.length; ++i) {
                        if (event.results[i].isFinal) {
                            finalTranscript += event.results[i][0].transcript;
                        }
                    }

                    if (finalTranscript) {
                        const currentText = chatInput.value;

                        updateSendButton();
                        chatInput.scrollTop = chatInput.scrollHeight;
                    }

            function findRelevantTransactions(query, history) {
                const lowerQuery = query.toLowerCase();
                const months = ['januari', 'februari', 'maret', 'april', 'mei', 'juni', 'juli', 'agustus', 'september', 'oktober', 'november', 'desember'];
                
                // 1. Cek Filter Waktu (Bulan/Tahun)
                let targetMonth = -1;
                let targetYear = null;
                
                months.forEach((m, i) => {
                    if (lowerQuery.includes(m)) targetMonth = i;
                });
                
                const yearMatch = lowerQuery.match(/\b20\d{2}\b/);
                if (yearMatch) targetYear = parseInt(yearMatch[0]);

                // 2. Cek Filter Kategori/Keyword
                const keywords = lowerQuery.split(' ').filter(w => w.length > 3);

                const filtered = history.filter(t => {
                    const tDate = new Date(t.date);
                    const tMonth = tDate.getMonth();
                    const tYear = tDate.getFullYear();
                    
                    let matchTime = true;
                    if (targetMonth !== -1 && tMonth !== targetMonth) matchTime = false;
                    if (targetYear !== null && tYear !== targetYear) matchTime = false;
                    
                    if (targetMonth === -1 && targetYear === null && lowerQuery.includes('bulan ini')) {
                        const now = new Date();
                        if (tMonth !== now.getMonth() || tYear !== now.getFullYear()) matchTime = false;
                    }

                    let matchKeyword = true;
                    if (targetMonth === -1 && targetYear === null && !lowerQuery.includes('bulan ini')) {
                        const content = (t.description + ' ' + t.category).toLowerCase();
                        const hasKeyword = keywords.some(k => content.includes(k));
                        if (keywords.length > 0 && !hasKeyword) matchKeyword = false;
                    }

                    return matchTime && matchKeyword;
                });

                if (filtered.length === 0 || lowerQuery.length < 3) {
                    return history.slice(0, 50); // Fallback ke 50 transaksi terakhir
                }
                
                return filtered;
            }

        function setupRealtimeListener(userId) {
            if (unsubscribeSnapshot) unsubscribeSnapshot();

            const q = collection(db, 'artifacts', appId, 'users', userId, 'transactions');

            unsubscribeSnapshot = onSnapshot(q, (snapshot) => {
                const transactions = [];
                snapshot.forEach((doc) => {
                    transactions.push({ id: doc.id, ...doc.data() });
                });

                transactionHistory = transactions.sort((a, b) => {
                    const dateA = new Date(a.date + 'T' + a.time);
                    const dateB = new Date(b.date + 'T' + b.time);
                    return dateB - dateA;
                });

                renderAll();
            }, (error) => {
                console.error("Snapshot Error:", error);
                showToast("Gagal memuat data: " + error.message, "error");
            });
        }

                    datasets: [{
                        data: expenseData,
                        backgroundColor: (currentChartType === 'Expense' ? expenseColors : incomeColors).slice(0, expenseLabels.length),
                        borderWidth: 1,
                        borderColor: '#ffffff'
                    }]
                } : {
                    labels: ['Belum ada data'],
                    datasets: [{ data: [1], backgroundColor: ['#E5E7EB'] }] // Abu-abu
                };

                categoryChartInstance = new Chart(ctxCategory, {
                    type: 'doughnut',
                    data: chartDataConfig,
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: { position: 'right', labels: { boxWidth: 12, font: { size: 11 } } },
                            tooltip: { 
                                callbacks: {
                                    label: function(context) {
                                        if (expenseLabels.length === 0) return 'Belum ada data';
                                        return context.label + ': ' + formatCurrency(context.raw);
                                    }
                                }
                            }
                        }
                    }
                });
            }

            // Bar Data
            const last7Days = [];
            for (let i = 6; i >= 0; i--) {
                const d = new Date();
                d.setDate(d.getDate() - i);
                last7Days.push(d.toLocaleDateString('id-ID').split('/').reverse().join('-')); 
            }

            const dailyIncome = new Array(7).fill(0);
            const dailyExpense = new Array(7).fill(0);

            transactionHistory.forEach(t => {
                const index = last7Days.indexOf(t.date);
                if (index !== -1) {
                    if (t.type === 'Income') dailyIncome[index] += parseInt(t.amount);
                    else dailyExpense[index] += parseInt(t.amount);
                }
            });

            const dateLabels = last7Days.map(d => {
                const parts = d.split('-');
                return `${parts[2]}/${parts[1]}`;
            });

            // PERBAIKAN: Menambahkan pengecekan elemen trendChart
            const canvasTrend = document.getElementById('trendChart');
            
            if (canvasTrend) {
                const ctxTrend = canvasTrend.getContext('2d');

                if (trendChartInstance) {
                    trendChartInstance.destroy();
                }

                trendChartInstance = new Chart(ctxTrend, {
                    type: 'bar',
                    data: {
                        labels: dateLabels,
                        datasets: [
                            {
                                label: 'Pemasukan',
                                data: dailyIncome,
                                backgroundColor: '#10B981', 
                                borderRadius: 4
                            },
                            {
                                label: 'Pengeluaran',
                                data: dailyExpense,
                                backgroundColor: '#DC2626', 
                                borderRadius: 4
                            }
                        ]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        scales: {
                            y: { 
                                beginAtZero: true,
                                ticks: { 
                                    callback: function(value) { return value >= 1000 ? (value/1000) + 'k' : value; },
                                    font: { size: 10 }
                                }
                            },
                            x: {
                                ticks: { font: { size: 10 } }
                            }
                        },
                        plugins: {
                            legend: { position: 'top', labels: { boxWidth: 12, font: { size: 11 } } },
                            tooltip: {
                                callbacks: {
                                    label: function(context) {
                                        return context.dataset.label + ': ' + formatCurrency(context.raw);
                                    }
                                }
                            }
                        }
                    }
                });
            }
        }

    </script>
